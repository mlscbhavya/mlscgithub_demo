        function setRole(role) {
            selectedRole = role;
            document.querySelectorAll('.role-btn').forEach(btn => btn.classList.remove('active'));
            event.target.classList.add('active');
            document.getElementById('portal-title').innerText = `${role} Portal`;
        }

        function handleLogin(e) {
            e.preventDefault();
            const username = document.getElementById('username').value;
            
            // Populate Greeting details
            document.getElementById('display-user').innerText = username;
            document.getElementById('display-role').innerText = selectedRole;

            // Magical Transition Out of Login
            document.getElementById('login-container').style.transition = 'all 1s ease';
            document.getElementById('login-container').style.opacity = '0';
            document.getElementById('login-container').style.transform = 'scale(0.8)';
            
            setTimeout(() => {
                document.getElementById('login-container').classList.add('hidden');
                document.getElementById('universe-container').classList.remove('hidden');
            }, 1000);
        }

        function zoomToSolarSystem() {
            // Smooth zoom interpolation through Z-axis translation simulating space travel
            const container = document.getElementById('universe-container');
            container.style.transform = 'translateY(-100%) scale(1.1)';
        }

        function resetToGalaxy() {
            const container = document.getElementById('universe-container');
            container.style.transform = 'translateY(0%) scale(1)';
            hidePlanetData();
        }

        function showPlanetData(planetKey) {
            const data = cosmicData[planetKey];
            document.getElementById('panel-planet-name-facts').innerText = `${data.name} Core Facts`;
            
            const factsList = document.getElementById('facts-list');
            const mysteriesList = document.getElementById('mysteries-list');
            
            factsList.innerHTML = '';
            mysteriesList.innerHTML = '';

            data.facts.forEach(fact => {
                let li = document.createElement('li');
                li.innerText = fact;
                factsList.appendChild(li);
            });

            data.mysteries.forEach(mystery => {
                let li = document.createElement('li');
                li.innerText = mystery;
                mysteriesList.appendChild(li);
            });

            document.getElementById('info-panel').classList.add('active');
        }

        function hidePlanetData() {
            document.getElementById('info-panel').classList.remove('active');
        }
